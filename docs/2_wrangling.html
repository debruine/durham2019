6 Main Verbs

# load the data
eye <- read_csv("data/eye_descriptions.csv")
## Parsed with column specification:
## cols(
##   .default = col_character(),
##   user_id = col_double(),
##   age = col_double()
## )
## See spec(...) for full column specifications.

select()

eye2 <- select(eye, user_id, user_id, `t1`)
head(eye2)

filter()

eye3 <- filter(eye, age >= 18, age < 40)

head(eye3)
range(eye3$age)
## [1] 18.0 38.7

arrange()

eye4 <- arrange(eye3, sex, desc(age))

head(eye4)

mutate()

eye %>%
  filter(row_number() < 6) %>%
  select(user_id, sex, age) %>%
  mutate(year = 2019,
         age = round(age),
         birthyear = year - age,
         sex2 = recode(sex, "male" = "M", 
                            "female" = "F", 
                            .default = "X")
         )

summarise()

eye %>%
  mutate(age = ifelse(age > 100, NA, age)) %>%
  summarise(mean_age = mean(age, na.rm = TRUE),
          max_age = max(age, na.rm = TRUE))

group_by()

eye %>%
  mutate(age = ifelse(age > 100, NA, age)) %>%
  group_by(sex) %>%
  summarise(mean_age = mean(age, na.rm = TRUE),
          max_age = max(age, na.rm = TRUE))

Additional Verbs

rename()

eye %>%
  filter(row_number() < 6) %>%
  rename(face1 = t1)

distinct()

tibble(
  x = c(1,1,2,2),
  y = c(1,1,2,3)
) %>%
  distinct()

count()

eye %>%
  mutate(age = round(age)) %>%
  count(sex, age)

slice()

eye %>%
  arrange(age) %>%
  group_by(sex) %>%
  slice(1:3)

pull()

age <- pull(eye, age)
